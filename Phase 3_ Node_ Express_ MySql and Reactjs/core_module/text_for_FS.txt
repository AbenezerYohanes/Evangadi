Node.js FS рѕърїЂрѕЇ рѕЮріЋрІхріљрІЇ?
FS рѕЏрѕѕрЅх File System рѕЏрѕѕрЅх рѕ▓рѕєріЋрЇБ рЇІрІГрѕјрЅйріЋ ріЦріЊ рѕЏрІЇрїФрІјрЅйріЋ (directories) рѕѕрѕўрѕхрѕФрЅх рІерѕџрІФрѕхрЅйрѕЇ рІе Node.js рІІріЊ (core) рѕърїЂрѕЇ ріљрІЇрЇб рЇІрІГрѕјрЅйріЋ рѕЏріЋрЅарЅЦрЇБ рѕўрї╗рЇЇрЇБ рѕўрѕ░рѕерІЮ ріЦріЊ рѕїрѕјрЅй рѕхрѕФрІјрЅйріЋ рѕѕрѕўрѕхрѕФрЅх ріЦріЋрїарЅђрѕЮрЅарЅ│рѕѕріЋрЇб

FS рѕърїЂрѕЅріЋ рѕѕрѕўрїарЅђрѕЮ рѕўрїђрѕўрѕфрІФ рІѕрІ░ рЇЋрѕ«рїЇрѕФрѕЏрЅйріЋ рѕЏрѕЮрїБрЅх ріарѕѕрЅЦріЋрЇб

JavaScript

const fs = require('fs');
ріерІџрІФ рЅаріІрѕІ рІерѕърїЂрѕЅріЋ рІерЅ░рѕѕрІФрІЕ рЅ░рїЇрЅБрѕФрЅх (functions) рѕўрїарЅђрѕЮ ріЦріЋрЅйрѕІрѕѕріЋрЇб

­ЪЊќ рЇІрІГрѕЇ рѕЏріЋрЅарЅЦ (Reading a File)
рЇІрІГрѕјрЅйріЋ рѕѕрѕЏріЋрЅарЅЦ рѕЂрѕѕрЅх рІІріЊ рѕўріЋрїѕрІХрЅй ріарѕЅріЋрЇб ріЦріљрІџрѕЁрѕЮ asynchronous (рЅарЅ░рѕФ рІФрѕЇрѕєріљ) ріЦріЊ synchronous (рЅарЅ░рѕФ рІерѕєріљ) ріЊрЅИрІЇрЇб

1. Asynchronous рІўрІ┤: fs.readFile()
рІГрѕЁ рІўрІ┤ рЇІрІГрѕЅріЋ рѕ▓рІФріљрЅЦ рѕїрѕјрЅй рѕхрѕФрІјрЅйріЋ ріЦріЋрІ▓рѕ░рѕФ рѕхрѕѕрѕџрЇѕрЅЁрІх blocking ріарІФрІ░рѕГрїЇрѕЮрЇб рЇІрІГрѕЅріЋ ріаріЋрЅЦрЅд рѕ▓рїерѕГрѕх рІЇрїцрЅ▒ріЋ (error рІѕрІГрѕЮ data) рѕѕрѕЏрїЇріўрЅх ріаріЋрІх callback function ріЦріЋрѕ░рїарІІрѕѕріЋрЇб рІГрѕЁ рІўрІ┤ рЅаріарЅЦрІЏріЏрІЇ рІГрѕўрѕерїБрѕЇ рѕЮріГріЋрІФрЅ▒рѕЮ рІерЇЋрѕ«рїЇрѕФрѕЏрЅйріЋріЋ рЇЇрїЦріљрЅх (performance) ріарІГрЅђріЋрѕхрѕЮрЇб

ріарїѕрЅБрЅА (Syntax):

JavaScript

fs.readFile(path, [options], callback);
path: рІерѕЮріЊріљрЅарІЇ рЇІрІГрѕЇ рЅдрЅ│ (рѕѕрѕЮрѕ│рѕї: 'myFile.txt')рЇб

options: ріарѕЏрѕФрїГ ріљрІЇрЇБ рѕѕрѕЮрѕ│рѕї рЇІрІГрѕЅ рІерЅ░рЅђрѕўрїарЅарЅхріЋ encoding (ріЦріЋрІ░ 'utf8') рѕўрїЇрѕѕрїй ріЦріЋрЅйрѕІрѕѕріЋрЇб

callback: рЇІрІГрѕЅ рЅ░ріљрЅд рѕ▓рІФрѕЇрЅЁ рІерѕџрїарѕФрІЇ рЅ░рїЇрЅБрѕГ ріљрІЇрЇб рѕЂрѕѕрЅх ріљрїѕрѕ«рЅйріЋ рІГрІѕрѕхрІ│рѕЇ: (err, data)рЇб err рѕхрѕЁрЅ░рЅх рѕ▓рЇѕрїарѕГ рѕхрѕЁрЅ░рЅ▒ріЋ рІГрІГрІЏрѕЇрЇц data рІ░рїЇрѕъ рІерЇІрІГрѕЅріЋ рІГрІўрЅх рІГрІГрІЏрѕЇрЇб

рѕЮрѕ│рѕї:

JavaScript

const fs = require('fs');

fs.readFile('message.txt', 'utf8', (err, data) => {
  if (err) {
    console.error('рЇІрІГрѕЅріЋ рѕЏріЋрЅарЅЦ ріарѕЇрЅ░рЅ╗рѕѕрѕЮ:', err);
    return;
  }
  console.log('рІерЇІрІГрѕЅ рІГрІўрЅх:', data);
});

console.log('рІГрѕЁ рѕўрѕЇрІЋріГрЅх рЇІрІГрѕЅ ріерѕўріљрЅарЅА рЅарЇірЅх рѕірІѕрїБ рІГрЅйрѕІрѕЇрЇб');
рІерІЇрїцрЅ▒ рѕЏрЅЦрѕФрѕфрІФ:

readFile рЇІрІГрѕЅріЋ рѕЏріЋрЅарЅЦ рІГрїђрѕЮрѕГріЊ рІѕрІ▓рІФрІЇ рІѕрІ░ рЅђрїБрІЕ рі«рІх рІГрІўрѕІрѕЇрЇб рѕхрѕѕрІџрѕЁ 'рІГрѕЁ рѕўрѕЇрІЋріГрЅх рЇІрІГрѕЅ ріерѕўріљрЅарЅА рЅарЇірЅх рѕірІѕрїБ рІГрЅйрѕІрѕЇрЇб' рІерѕџрѕѕрІЇ рѕўрѕЇрІЋріГрЅх рѕўрїђрѕўрѕфрІФ рІГрІѕрїБрѕЇрЇб

рЇІрІГрѕЅ рЅ░ріљрЅд рѕ▓рІФрѕЇрЅЁрЇБ callback рЅ░рїЇрЅБрѕЕ рЅ░рїарѕГрЅХ рІерЇІрІГрѕЅріЋ рІГрІўрЅх рІѕрІГрѕЮ рѕхрѕЁрЅ░рЅ▒ріЋ рІФрѕ│рІЇрЅЃрѕЇрЇб

2. Synchronous рІўрІ┤: fs.readFileSync()
рІГрѕЁ рІўрІ┤ рЇІрІГрѕЅріЋ ріаріЋрЅЦрЅд ріЦрѕхріфрїерѕГрѕх рІхрѕерѕх рЇЋрѕ«рїЇрѕФрѕЎріЋ рІФрЅєрѕўрІІрѕЇ (blocks)рЇб рІерЇІрІГрѕЅріЋ рІГрІўрЅх рЅарЅђрїЦрЅ│ рІГрѕўрѕЇрѕ│рѕЇрЇб рІГрѕЁ рІўрІ┤ рІерЇЋрѕ«рїЇрѕФрѕЏрЅйріЋріЋ рЇЇрїЦріљрЅх рѕхрѕѕрѕџрЅђріЋрѕх рЅаріарЅЦрІЏріЏрІЇ ріарІГрѕўріерѕГрѕЮрЇБ рїЇріЋ рЅђрѕІрѕЇ ріЦріЊ рЇѕрїБріЋ рѕхрѕФрІјрЅйріЋ рѕѕрѕўрѕхрѕФрЅх рѕірїарЅЁрѕЮ рІГрЅйрѕІрѕЇрЇб

ріарїѕрЅБрЅА (Syntax):

JavaScript

const data = fs.readFileSync(path, [options]);
path: рІерѕЮріЊріљрЅарІЇ рЇІрІГрѕЇ рЅдрЅ│рЇб

options: ріарѕЏрѕФрїГ ріљрІЇрЇБ рѕѕрѕЮрѕ│рѕї encoding рѕўрїЇрѕѕрїйрЇб

data: рІерЇІрІГрѕЅріЋ рІГрІўрЅх рЅарЅђрїЦрЅ│ рІГрІГрІЏрѕЇрЇб

рѕЮрѕ│рѕї:

JavaScript

const fs = require('fs');

try {
  const data = fs.readFileSync('message.txt', 'utf8');
  console.log('рІерЇІрІГрѕЅ рІГрІўрЅх:', data);
} catch (err) {
  console.error('рЇІрІГрѕЅріЋ рѕЏріЋрЅарЅЦ ріарѕЇрЅ░рЅ╗рѕѕрѕЮ:', err);
}

console.log('рІГрѕЁ рѕўрѕЇрІЋріГрЅх рІерѕџрІѕрїБрІЇ рЇІрІГрѕЅ ріерЅ░ріљрЅарЅа рЅаріІрѕІ рЅЦрЅ╗ ріљрІЇрЇб');
рІерІЇрїцрЅ▒ рѕЏрЅЦрѕФрѕфрІФ:

readFileSync рЇІрІГрѕЅріЋ ріаріЋрЅЦрЅд ріЦрѕхріфрїерѕГрѕх рІхрѕерѕх рІерЇЋрѕ«рїЇрѕФрѕЎ ріарЇѕрЇЃрЇђрѕЮ рІГрЅєрѕЏрѕЇрЇб

рІерЇІрІГрѕЅ рІГрІўрЅх data рЅарЅ░рЅБрѕѕрІЇ рЅ░рѕѕрІІрІІрїГ рІЇрѕхрїЦ ріерїѕрЅБ рЅаріІрѕІ ріљрІЇ рЅђрїБрІЕ рі«рІх рІерѕџрѕ░рѕФрІЇрЇб

рѕхрѕЁрЅ░рЅх рѕірЇѕрїарѕГ рѕхрѕѕрѕџрЅйрѕЇ рі«рІ▒ріЋ рЅаtry...catch block рІЇрѕхрїЦ рѕЏрѕхрЅђрѕўрїЦ рІГрѕўріерѕФрѕЇрЇб

­ЪњА рѕЏрїарЅЃрѕѕрІФ
fs.readFile(): Asynchronous ріљрІЇрЇБ рЅарЇЇрїЦріљрЅх рІГрѕ░рѕФрѕЇрЇБ callback рІГрїарЅђрѕЏрѕЇрЇБ рІГрѕўріерѕФрѕЇрЇб

fs.readFileSync(): Synchronous ріљрІЇрЇБ рЇЋрѕ«рїЇрѕФрѕЎріЋ рІФрїЇрІ│рѕЇрЇБ рІЇрїцрЅ▒ріЋ рЅарЅђрїЦрЅ│ рІГрѕўрѕЇрѕ│рѕЇрЇБ рЅаріарЅЦрІЏріЏрІЇ ріарІГрѕўріерѕГрѕЮ (рЅарїЦріЋрЅЃрЅё рїЦрЅЁрѕЮ рѕІрІГ рѕўрІІрѕЇ ріарѕѕрЅарЅх)рЇб
